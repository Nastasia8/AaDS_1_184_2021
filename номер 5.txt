def privet(a, b):
    e = a*b
    while a != 0 and b != 0:
        if a > b:
            a %= b 
        else:
            b %= a 
    return e//(a + b)
    
a = int(input("a= "))
b = int(input("b= "))
print("PRIVET(", a, ";", b, ") = ", privet(a, b))
