def funct(n, tup):
    if n in tup:
      start= tup.index(n)
      k=tup.count(n)
      if k==1:
        return(tup[start:])
      else:
        i=1
        end=start
        while i<k:
          end=tup.index(n,end+1)
          i+=1
          return tup[start:end+1]
      return tup[start]    
    else:
      print("no element in",tup)
