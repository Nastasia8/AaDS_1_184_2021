def xox(l):
    if l == 0:
        return 0
    elif l == 1:
        return 3
    elif l == 2:
        return 5
    else:
        return xox(l - 1) + xox(l - 2) + xox(l - 3)
for  i in range(1, 11):
    print("a[", i, "]=" , xox(i))
